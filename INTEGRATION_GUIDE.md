### 1. Update Coordinator to Use Naming & Style Config

**File:** `pystoorm/generator/coordinator.py`

Currently, the Coordinator generates code without considering naming conventions or style guides.

**What needs to change:**

```python
class Coordinator:
    def __init__(self, config):
        self.config = config
        self.naming = config.get('naming', {})
        self.style = config.get('style', {})
        self.hints = config.get('attribute_hints', {})

    def _transform_class_name(self, table_name):
        """Transform table name to class name using style config"""
        case = self.style['class_naming']['case']
        suffix = self.style['class_naming']['suffix']
        prefix = self.style['class_naming']['prefix']

        # Transform case
        transformed = self._apply_case_transform(table_name, case)
        # Apply prefix/suffix
        return f"{prefix}{transformed}{suffix}"

    def _get_attribute_hint(self, table_name, table):
        """Find best display attribute for table"""
        hints_config = self.hints

        # Check table-specific overrides
        if table_name in hints_config.get('overrides', {}):
            candidates = hints_config['overrides'][table_name]
        else:
            candidates = hints_config.get('defaults', [])

        # Find first matching column
        for candidate in candidates:
            col_name = self._expand_pattern(candidate, table_name)
            if col_name in table.columns:
                return col_name

        # Fallback
        for fallback in hints_config.get('fallback', ['id']):
            if fallback in table.columns:
                return fallback

        return None

    def _is_primary_key(self, column, table_name):
        """Check if column matches primary key pattern"""
        patterns = self.naming['primary_key'].get('patterns', [])
        for pattern in patterns:
            if self._matches_pattern(pattern, column.name, table_name):
                return True
        return False

    def _matches_pattern(self, pattern, column_name, table_name):
        """Check if column name matches naming pattern"""
        # Expand patterns like {table_singular} or {other_table}
        expanded = pattern.replace('{table_singular}', table_name.rstrip('s'))
        # Simple regex matching
        import re
        return bool(re.match(f"^{expanded}$", column_name))
```

### 2. Update Templates to Use Generated Metadata

**File:** `pystoorm/templates/python/model.py`

```mako
## modus: table
<%
    # Get configuration
    style = context.get('style', {})
    hints = context.get('attribute_hints', {})

    # Transform class name based on style
    class_name = transform_class_name(table_name, style)

    # Get display attribute
    display_attr = get_attribute_hint(table_name, hints)
%>

# Generated by pyStoOrm
from sqlalchemy.orm import declarative_base
from datetime import datetime

Base = declarative_base()

class ${class_name}(Base):
    """ORM Model for table: ${table_name}"""
    __tablename__ = "${table_name}"

    % for column in table.columns.values():
    ${column.name}: ${get_python_type(column)} = Column(...)
    % endfor

    def __str__(self):
        % if display_attr:
        return str(self.${display_attr})
        % else:
        return f"${class_name}(id={self.id})"
        % endif

    def __repr__(self):
        return f"<${class_name}({self.id})>"
```

### 3. Pass Configuration to Templates

**File:** `pystoorm/generator/coordinator.py`

```python
def generate(self):
    for template in self.config['output']:
        if not template.get('enabled', True):
            continue

        if template['modus'] == 'table':
            for connection in self.config['connections']:
                schema = connection['parsedSchema']
                for table_name in schema.table_names:
                    table = schema.tables[table_name]

                    # Create context with configuration
                    ctx = {
                        'table': table,
                        'table_name': table_name,
                        'schema': schema,

                        # Add configuration to context
                        'style': self.config['style'],
                        'naming': self.config['naming'],
                        'attribute_hints': self.config['attribute_hints'],

                        # Add helper functions
                        'transform_class_name': self._transform_class_name,
                        'get_attribute_hint': self._get_attribute_hint,
                        'is_primary_key': self._is_primary_key,
                        'is_foreign_key': self._is_foreign_key,
                        'get_python_type': self._get_python_type,
                    }

                    self._render_template(template, ctx)
```

### 5. Configuration File Organization Example

For a large project, you might organize configurations like this:

```
my_project/
├── project.yml                     # Main connection + overrides
├── config/
│   ├── naming-conventions.yml      # Custom naming rules
│   ├── style-guides/
│   │   └── my-company-standard.yml # Company style guide
│   └── attribute-hints.yml         # Custom display hints
└── src/
    └── orm/
        ├── models/
        └── repositories/
```

**In `project.yml`:**
```yaml
connections:
  - connection: main
    host: localhost
    database: myapp
    connector: database.mysqlconnector.MysqlConnector

# These will be loaded from the files above
naming: &naming_config
  $ref: config/naming-conventions.yml

style: &style_config
  $ref: config/style-guides/my-company-standard.yml
```

*Note: YAML doesn't support $ref natively; you'd need to load these manually or use a library like `refs`*

### 6. Usage in Your Application

Once configured and integrated:

```python
from pystoorm.config.loader import load_config
from pystoorm.analyzer.controller import Controller
from pystoorm.generator.coordinator import Coordinator

# Load configuration
config = load_config("my_project/project.yml")

# Analyze database
analyzer = Controller(config)
analyzer.walk()

# Generate code with configuration applied
coordinator = Coordinator(config)
coordinator.generate()

# Generated code will now:
# ✓ Use correct naming conventions
# ✓ Follow your style guide
# ✓ Display correct attributes in dropdowns
```

## 💡 Code Generation Example

Once integration is complete, here's how it will work:

**Config:**
```yaml
naming:
  primary_key:
    patterns: ["^id$", "^{table_singular}_id$"]

style:
  class_naming:
    suffix: "Model"
    case: "PascalCase"

attribute_hints:
  defaults: ["name"]
  overrides:
    users: ["last_name"]
```

**Database:**
```sql
CREATE TABLE users (
    id INT PRIMARY KEY,
    last_name VARCHAR(50),
    first_name VARCHAR(50)
);
```

**Generated Code:**
```python
class UsersModel(Base):
    __tablename__ = 'users'
    id: int = Column(...)
    last_name: str = Column(...)
    first_name: str = Column(...)

    def __str__(self):
        return str(self.last_name)  # Uses configured hint
```
